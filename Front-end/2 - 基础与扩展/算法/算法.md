        function sort(arr) {
            let len = arr.length
            for (let i = 0; i < len; i++) {
                for (let j = i+1; j < len; j++) {
                    if (arr[j] > arr[i]) {
                        [arr[i], arr[j]] = [arr[j], arr[i]]
                    }
                }
            }
